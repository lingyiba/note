  [^list] #匹配 除 list 中的任意单一字符以外的字符
  [c1-c2] #匹配 c1-c2 中的任意单一字符，如：[0-9][a-z]
  {string1,string2,...} #匹配 c1-c2 中全部字符，如：{1..10}
